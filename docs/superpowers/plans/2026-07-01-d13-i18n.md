# D13 — EN/ES Internationalization Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Make the Herrera public site fully bilingual (English default at the root, Spanish under `/es/`) with proper multilingual SEO, and make Nilyan's admin-authored content (qualification questions, guides, off-market listings) enterable and rendered in both languages.

**Architecture:** Next.js Pages-Router **built-in i18n routing** (`locales: ['en','es']`, `defaultLocale: 'en'`, no prefix on `en`, `/es` prefix on `es`). Static UI strings come from a **lightweight typed message dictionary** (`en.ts` defines the canonical `Messages` type; `es.ts` must satisfy it, so a missing translation is a **compile error**) read through a `useTranslation()` hook keyed off `useRouter().locale`. Admin-authored content is localized **server-side** in `getStaticProps`/`getServerSideProps` using `context.locale`, so the ES page ships ES HTML (SEO + no flash), falling back to EN when an ES value is empty. SEO (`hreflang` + `x-default` + self-referential canonical + bilingual sitemap) is wired into the existing `<Seo>` seam and the sitemap builder.

**Tech Stack:** Next.js 15.1 (Pages Router), React 19, TypeScript (strict), Drizzle ORM + Postgres/Neon, Zod, vitest. **No new i18n runtime dependency** — the dictionary is plain typed TS.

## Global Constraints

Copied verbatim from ADR-018 (the approved spec) + the locked scope memory `herrera-d13-i18n`. Every task's requirements implicitly include this section.

- **Locales:** `['en','es']`, `defaultLocale: 'en'`. **English at the root, no prefix** (`/search`); **Spanish under `/es/`** (`/es/search`). Two indexable URLs per public page.
- **`localeDetection: false`** — do NOT auto-redirect on `Accept-Language`. The root stays English and predictable (SEO). The header toggle is the only locale switch.
- **Silent, graceful EN fallback** — when an ES value is empty, render the English value. **Never** show a visitor-facing "missing translation" placeholder, an empty gap, or broken layout.
- **Multilingual SEO (top priority):** reciprocal `hreflang` alternates + `x-default` on **every** public page; **bilingual sitemap** (both language URLs of every page); correct **`<html lang>`** per locale; **self-referential canonical** per locale (each locale canonical to itself, not the other).
- **Header EN/ES toggle** switches to the **same** page in the other language, preserving route + query.
- **No heavy i18n dependency** — a typed TS dictionary + `useTranslation()` hook. Do not add `next-i18next`, `next-intl`, `i18next`, or `react-intl`.
- **Schema — only these ES columns are NEW** (add via one migration): `listings.description_es`, `content.meta_title_es`, `content.meta_description_es`. **Reuse existing** ES columns (do NOT re-add): `content.title_es / excerpt_es / body_es`, `qualification_questions.label_es` (+ per-option `labelEs` in the options JSONB), `search_filters.label_es`.
- **MLS property-description translation is DEFERRED** (post-feed) — out of D13. Only `source='manual'` (off-market) descriptions get `description_es`.
- **Admin UI chrome stays English-only** — only the **content fields** Nilyan authors become bilingual, not the admin interface.

```ts
export const LOCALES = ["en", "es"] as const;
export type Locale = (typeof LOCALES)[number];
export const DEFAULT_LOCALE: Locale = "en";

export function isLocale(x: unknown): x is Locale {
  return typeof x === "string" && (LOCALES as readonly string[]).includes(x);
}

/** Next may pass `locale` as string | undefined; normalize to a supported Locale. */
export function asLocale(x: unknown): Locale {
  return isLocale(x) ? x : DEFAULT_LOCALE;
}

/** Localized value with a silent EN fallback: ES only when present + non-blank under `es`. */
export function pickLocalized(
  base: string,
  es: string | null | undefined,
  locale: Locale,
): string {
  if (locale === "es" && typeof es === "string" && es.trim() !== "") return es;
  return base;
}
```

```ts
import { DEFAULT_LOCALE, LOCALES, type Locale } from "@/lib/i18n/config";

/** Absolute URL for a site path in a given locale (en = no prefix, es = "/es"). */
export function localizedUrl(path: string, locale: Locale): string {
  const clean = path === "/" ? "" : path;
  if (locale === DEFAULT_LOCALE) return absoluteUrl(path);
  return `${SITE_URL}/${locale}${clean}`;
}

export type AlternateLink = { hrefLang: string; href: string };

/** Reciprocal hreflang set + x-default (= default locale) for a site path. */
export function alternatesFor(path: string): AlternateLink[] {
  const links: AlternateLink[] = LOCALES.map((l) => ({ hrefLang: l, href: localizedUrl(path, l) }));
  return [...links, { hrefLang: "x-default", href: localizedUrl(path, DEFAULT_LOCALE) }];
}

/** Bilingual sitemap: two <url> entries per path, each carrying the xhtml alternates. */
export function buildSitemapXml(paths: string[]): string {
  const alt = (path: string) =>
    LOCALES.map((l) => `    <xhtml:link rel="alternate" hreflang="${l}" href="${localizedUrl(path, l)}"/>`)
      .concat(`    <xhtml:link rel="alternate" hreflang="x-default" href="${localizedUrl(path, DEFAULT_LOCALE)}"/>`)
      .join("\n");
  const entries = paths
    .flatMap((path) =>
      LOCALES.map(
        (l) => `  <url>\n    <loc>${localizedUrl(path, l)}</loc>\n${alt(path)}\n  </url>`,
      ),
    )
    .join("\n");
  return `<?xml version="1.0" encoding="UTF-8"?>\n<urlset xmlns="http://www.sitemaps.org/schemas/sitemap/0.9" xmlns:xhtml="http://www.w3.org/1999/xhtml">\n${entries}\n</urlset>\n`;
}
```

```tsx
import Head from "next/head";
import { useRouter } from "next/router";
import { absoluteUrl, alternatesFor, localizedUrl, SITE_NAME } from "@/lib/seo";
import { asLocale } from "@/lib/i18n/config";

type Props = {
  title: string;
  description?: string;
  path?: string;
  image?: string | null;
  noindex?: boolean;
  jsonLd?: object | object[];
};

/** Reusable head: title, description, per-locale canonical, hreflang alternates, OG/Twitter, JSON-LD. */
export function Seo({ title, description, path, image, noindex, jsonLd }: Props) {
  const locale = asLocale(useRouter().locale);
  const canonical = path ? localizedUrl(path, locale) : undefined;
  const alternates = path ? alternatesFor(path) : [];
  const blocks = jsonLd ? (Array.isArray(jsonLd) ? jsonLd : [jsonLd]) : [];
  return (
    <Head>
      <title>{title}</title>
      {description && <meta name="description" content={description} />}
      {noindex && <meta name="robots" content="noindex,nofollow" />}
      {canonical && <link rel="canonical" href={canonical} />}
      {alternates.map((a) => (
        <link key={a.hrefLang} rel="alternate" hrefLang={a.hrefLang} href={a.href} />
      ))}

      <meta property="og:type" content="website" />
      <meta property="og:site_name" content={SITE_NAME} />
      <meta property="og:locale" content={locale === "es" ? "es_ES" : "en_US"} />
      <meta property="og:title" content={title} />
      {description && <meta property="og:description" content={description} />}
      {canonical && <meta property="og:url" content={canonical} />}
      {image && <meta property="og:image" content={image} />}

      <meta name="twitter:card" content={image ? "summary_large_image" : "summary"} />
      <meta name="twitter:title" content={title} />
      {description && <meta name="twitter:description" content={description} />}
      {image && <meta name="twitter:image" content={image} />}

      {blocks.map((block, i) => (
        <script
          key={i}
          type="application/ld+json"
          dangerouslySetInnerHTML={{ __html: JSON.stringify(block) }}
        />
      ))}
    </Head>
  );
}
```

Note: `absoluteUrl` import retained only if still referenced; if lint flags it unused, drop it from the import.

```ts
import { describe, expect, it } from "vitest";
import { en } from "./messages/en";
import { es } from "./messages/es";

function keyPaths(obj: unknown, prefix = ""): string[] {
  if (obj && typeof obj === "object" && !Array.isArray(obj)) {
    return Object.entries(obj as Record<string, unknown>).flatMap(([k, v]) =>
      keyPaths(v, prefix ? `${prefix}.${k}` : k),
    );
  }
  return [prefix];
}

describe("message dictionaries", () => {
  it("es has exactly the same keys as en (no gaps, no extras)", () => {
    expect(keyPaths(es).sort()).toEqual(keyPaths(en).sort());
  });
  it("no es value is an empty string", () => {
    const empties = keyPaths(es).filter((p) =>
      p.split(".").reduce<any>((o, k) => o?.[k], es) === "",
    );
    expect(empties).toEqual([]);
  });
});
```
